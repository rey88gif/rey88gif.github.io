GoRouter get router {
  if (_router != null) {
    return _router!;
  }

  final routes = [
    GoRoute(
      name: Routes.login,
      path: '/login',
      builder: (_, __) => const LoginView(),
    ),
    GoRoute(
      name: Routes.settingDatabaseStorage,
      path: '/setting-database-storage',
      builder: (_, __) => const SettingDatabaseStorageView(),
    ),
    GoRoute(
      name: Routes.home,
      path: '/',
      builder: (_, __) => const HomeView(),
    ),
    GoRoute(
      name: Routes.register,
      path: '/register',
      builder: (_, state) => const RegisterView(),
    ),
    GoRoute(
      name: Routes.history,
      path: '/history',
      builder: (_, __) => const HistoryView(),
    ),
    GoRoute(
      name: Routes.employees,
      path: '/employees',
      builder: (_, __) => const EmployeesView(),
    ),
    GoRoute(
      name: Routes.addEmployee,
      path: '/add-employee',
      builder: (_, __) => const EmployeeView(),
    ),
    GoRoute(
      name: Routes.editEmployee,
      path: '/edit-employee/:id',
      builder: (_, state) => EmployeeView(
        id: int.parse(state.pathParameters['id']!),
      ),
    ),
    GoRoute(
      name: Routes.products,
      path: '/products',
      builder: (_, __) => const ProductsView(),
    ),
    GoRoute(
      name: Routes.addProduct,
      path: '/add-product',
      builder: (_, __) => const ProductView(),
    ),
    GoRoute(
      name: Routes.coins,
      path: '/coins',
      builder: (_, __) => const CoinsView(),
    ),
    GoRoute(
      name: Routes.addCoin,
      path: '/add-coin',
      builder: (_, __) => const CoinView(),
    ),
    // GoRoute(
    //   name: Routes.editCoin,
    //   path: '/edit-coin/:short_name',
    //   builder: (_, state) => CoinView(
    //     shortName: state.pathParameters['short_name'],
    //   ),
    // ),
    GoRoute(
      name: Routes.paymentMethods,
      path: '/payment-methods',
      builder: (_, __) => const PaymentMethodsView(),
    ),
    GoRoute(
      name: Routes.export,
      path: '/export',
      builder: (_, __) => const ExportView(),
    ),
    GoRoute(
      name: Routes.about,
      path: '/about',
      builder: (_, __) => const AboutView(),
    ),
    GoRoute(
      name: Routes.discounts,
      path: '/discounts',
      builder: (_, __) => const DiscountsView(),
    ),
    GoRoute(
      name: Routes.billAdd,
      path: '/bill-add',
      builder: (_, __) => const AddBillView(),
    ),
    GoRoute(
      name: Routes.billPay,
      path: '/bill-pay',
      builder: (_, __) => const BillPayView(),
    ),
  ];

  final overrideRoutes = widget.overrideRoutes;
  if (overrideRoutes?.isNotEmpty ?? false) {
    final names = overrideRoutes!.map(
      (e) => e.name,
    );
    routes.removeWhere(
      (element) {
        final name = element.name;
        if (name != null) {
          return names.contains(name);
        }
        return false;
      },
    );
    routes.addAll(overrideRoutes);
  }

  final initialLocation = routes
      .firstWhere(
        (element) => element.name == _initialRouteName,
        orElse: () => routes.first,
      )
      .path;

  _router = GoRouter(
    initialLocation: initialLocation,
    routes: routes,
  );
  return _router!;
}
