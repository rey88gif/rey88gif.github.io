Describes the part of the user interface represented by this widget.

The framework calls this method when this widget is inserted into the tree in a given BuildContext and when the dependencies of this widget change (e.g., an InheritedWidget referenced by this widget changes). This method can potentially be called in every frame and should not have any side effects beyond building a widget.

The framework replaces the subtree below this widget with the widget returned by this method, either by updating the existing subtree or by removing the subtree and inflating a new subtree, depending on whether the widget returned by this method can update the root of the existing subtree, as determined by calling Widget.canUpdate.

Typically implementations return a newly created constellation of widgets that are configured with information from this widget's constructor and from the given BuildContext.

The given BuildContext contains information about the location in the tree at which this widget is being built. For example, the context provides the set of inherited widgets for this location in the tree. A given widget might be built with multiple different BuildContext arguments over time if the widget is moved around the tree or if the widget is inserted into the tree in multiple places at once.

The implementation of this method must only depend on:

  • the fields of the widget, which themselves must not change over time, and
  • any ambient state obtained from the context using BuildContext.dependOnInheritedWidgetOfExactType.

If a widget's build method is to depend on anything else, use a StatefulWidget instead.

See also:

  • StatelessWidget, which contains the discussion on performance considerations.

@override
Widget build(BuildContext context) {
  String oldDiscount = '0.00';
  final code = ''.obs;
  final productController = Get.put(ProductsController());
  final coinsController = Get.put(CoinController()..getCoins());
  final Rxn<Product> product = Rxn();

  final isVisible = true.obs;
  final productCodeController = TextEditingController().obs;
  final productNameController = TextEditingController().obs;
  final productStockController = TextEditingController(text: '0').obs;
  final productPriceController = TextEditingController(text: '0.00').obs;
  final productDiscountController = TextEditingController(text: '0.00').obs;

  final stockFocusNode = FocusNode().obs;
  final priceFocusNode = FocusNode().obs;
  final discountFocusNode = FocusNode().obs;

  stockFocusNode.value.addIntTextFieldListener(productStockController.value);
  priceFocusNode.value
      .addDoubleTextFieldListener(productPriceController.value);
  discountFocusNode.value
      .addDoubleTextFieldListener(productDiscountController.value);

  final theme = Theme.of(context);
  final textTheme = theme.textTheme;
  final screenSize = MediaQuery.sizeOf(context);
  final dataRepository = Repositories.data;
  final account = dataRepository.account;

  if (Get.arguments != null) {
    product.value = Get.arguments['product'] as Product;
    productCodeController.value.text = product.value!.code;
    productNameController.value.text = product.value!.name;
    productStockController.value.text = product.value!.stock.toString();
    productPriceController.value.text = product.value!.price.toString();
    productDiscountController.value.text = product.value!.discount.toString();
    isVisible.value = product.value!.isVisible;
  }

  Product initProduct() => Product(
        code: productCodeController.value.text.trimRight(),
        name: productNameController.value.text.trimRight(),
        stock: int.parse(productStockController.value.text),
        price: double.parse(productPriceController.value.text),
        coinType: DropdownButtonCoinType.value,
        isVisible: isVisible.value,
        createdBy: account!.username,
        discount: (productDiscountController.value.text != '')
            ? double.parse(productDiscountController.value.text)
            : 0,
      );

  void requestBack() {
    showDialog(
      context: context,
      builder: (context) => WarningDialog(
        warning: '¿Está seguro que desea descartar los cambios?',
        onAccept: () async {
          await productController.getProducts();
          Get.offAndToNamed(Routes.products);
        },
      ),
    );
  }

  return OrientationBuilder(builder: (context, orientation) {
    final isLandscape = orientation == Orientation.landscape;
    return PopScope(
      canPop: false,
      onPopInvoked: (didPop) {
        if (didPop) return;
        requestBack();
      },
      child: Scaffold(
        appBar: MyAppBar(
          title: 'Productos',
          onBack: () async {
            requestBack();
          },
          actions: [
            IconButton(
                onPressed: () {
                  showMaterialModalBottomSheet(
                    context: context,
                    builder: (context) {
                      final mobileScannerController =
                          MobileScannerController();

                      final double scanWindowWidth =
                          MediaQuery.of(context).size.width * 0.6;
                      final double scanWindowHeight =
                          MediaQuery.of(context).size.height * 0.2;
                      final double scanWindowLeft =
                          (MediaQuery.of(context).size.width -
                                  scanWindowWidth) /
                              2; // Centra horizontalmente
                      final double scanWindowTop =
                          (MediaQuery.of(context).size.height -
                                  scanWindowHeight) /
                              2;

                      return Scaffold(
                        body: Stack(children: [
                          MobileScanner(
                            controller: mobileScannerController,
                            scanWindow: Rect.fromLTWH(
                              scanWindowLeft,
                              scanWindowTop,
                              scanWindowWidth,
                              scanWindowHeight,
                            ),
                            onDetect: (barcodes) async {
                              if (barcodes.raw != null) {
                                final codeList = barcodes.raw! as List;
                                final codeMap = codeList.first as Map;
                                code.value = codeMap['rawValue'].toString();
                                productCodeController.value.text = code.value;
                                await mobileScannerController.dispose();
                                Get.back();
                              }
                            },
                          ),
                          Positioned(
                            left: scanWindowLeft,
                            top: scanWindowTop,
                            child: Container(
                              width: scanWindowWidth,
                              height: scanWindowHeight,
                              decoration: BoxDecoration(
                                border:
                                    Border.all(color: Colors.grey, width: 3),
                                borderRadius: BorderRadius.circular(12),
                              ),
                            ),
                          ),
                          Positioned(
                            left: scanWindowLeft,
                            top: scanWindowTop + scanWindowHeight + 10,
                            child: Obx(
                              () => Text(
                                code.value == ''
                                    ? 'Enfoca el código aquí'
                                    : productController.productCode.value,
                                style: TextStyle(
                                    color: Colors.grey, fontSize: 16),
                              ),
                            ),
                          ),
                        ]),
                      );
                    },
                  );
                  //Get.toNamed(Routes.barcodeScanner);
                },
                icon: Icon(Icons.barcode_reader))
          ],
        ),
        body: SingleChildScrollView(
          child: Padding(
            padding: isLandscape
                ? EdgeInsets.symmetric(
                    horizontal: screenSize.width * 0.1, vertical: 10)
                : const EdgeInsets.symmetric(horizontal: 28),
            child: Column(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                if (isLandscape) const SizedBox(height: 10),
                Column(
                  children: [
                    Row(
                      mainAxisAlignment: MainAxisAlignment.end,
                      children: [
                        Text(
                          'Visible',
                          style: textTheme.bodyMedium?.copyWith(
                            fontWeight: FontWeight.w500,
                          ),
                        ),
                        Obx(
                          () => Checkbox(
                              visualDensity: VisualDensity.compact,
                              value: isVisible.value,
                              onChanged: (value) => isVisible.value = value!),
                        ),
                      ],
                    ),
                    MyTextField(
                      textCapitalization: TextCapitalization.sentences,
                      inputFormatters: [
                        FilteringTextInputFormatter.allow(
                            RegExp(r'[a-zA-Z0-9 áéíóúÁÉÍÓÚÑñ]')),
                      ],
                      controller: productNameController.value,
                      labelText: 'Nombre del producto*',
                      onChanged: (value) {
                        if (productNameController.value.text
                            .startsWith(' ')) {
                          String newText =
                              productNameController.value.text.trimLeft();
                          productNameController.value.text = newText;

                          productNameController.value.selection =
                              TextSelection.fromPosition(
                            TextPosition(offset: newText.length),
                          );
                        }
                      },
                    ),
                    const SizedBox(
                      height: 16,
                    ),
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceBetween,
                      crossAxisAlignment: CrossAxisAlignment.end,
                      children: [
                        Expanded(
                          flex: 1,
                          child: MyTextField(
                            onTap: () {
                              if (productPriceController.value.text ==
                                  '0.00') {
                                productPriceController.value.clear();
                              }
                            },
                            onChanged: (value) {},
                            focusNode: priceFocusNode.value,
                            controller: productPriceController.value,
                            labelText: 'Precio del producto*',
                            keyboardType: TextInputType.numberWithOptions(
                                decimal: true),
                            inputFormatters: [
                              FilteringTextInputFormatter.allow(
                                RegExp(r'^\d+\.?\d{0,2}'),
                              ),
                            ],
                            // TODO: Coins
                          ),
                        ),
                        SizedBox(
                          width: 48,
                        ),
                        SizedBox(
                          width: 60,
                          child: Obx(
                            () => DropdownButtonCoinType(
                              textTheme: textTheme,
                              items: coinsController.coins.value
                                  ?.map((e) => e.shortName)
                                  .toList(),
                            ),
                          ),
                        ),
                      ],
                    ),
                    const SizedBox(height: 16),
                    MyTextField(
                      controller: productStockController.value,
                      labelText: 'Cantidad en existencia*',
                      maxLength: 10,
                      focusNode: stockFocusNode.value,
                      onTap: () {
                        if (productStockController.value.text == '0') {
                          productStockController.value.clear();
                        }
                      },
                      keyboardType: TextInputType.number,
                      inputFormatters: [
                        FilteringTextInputFormatter.allow(
                          RegExp(r'^\d+'),
                        )
                      ],
                    ),
                    const SizedBox(height: 16),
                    MyTextField(
                      controller: productCodeController.value,
                      maxLength: 30,
                      inputFormatters: Validations.alphaNumericNotSpaces(),
                      readOnly: product.value?.code != null,
                      labelText: 'Código del producto*',
                      onChanged: (value) {
                        if (productCodeController.value.text
                            .startsWith(' ')) {
                          String newText =
                              productCodeController.value.text.trimLeft();
                          productCodeController.value.text = newText;

                          productCodeController.value.selection =
                              TextSelection.fromPosition(
                                  TextPosition(offset: newText.length));
                        }
                      },
                    ),
                    MyTextField(
                      onTap: () {
                        if (productDiscountController.value.text == '0.00') {
                          productDiscountController.value.clear();
                        }
                      },
                      onChanged: (value) {
                        if (value != '' && double.parse(value) > 99.99) {
                          MyToast.showToast(MyToastType.error,
                              "El descuento no puede exceder 99.99%");
                          productDiscountController.value.text = oldDiscount;
                          return;
                        }
                        oldDiscount = value;
                      },
                      controller: productDiscountController.value,
                      labelText: 'Descuento (%)',
                      focusNode: discountFocusNode.value,
                      keyboardType:
                          TextInputType.numberWithOptions(decimal: true),
                      inputFormatters: [
                        FilteringTextInputFormatter.allow(
                          RegExp(r'^(100|\d{1,2})\.?\d{0,2}'),
                        ),
                      ],
                    ),
                  ],
                ),
                SizedBox(height: isLandscape ? 20 : 32),
                Padding(
                  padding: isLandscape
                      ? EdgeInsets.symmetric(
                          horizontal: screenSize.width * 0.15)
                      : EdgeInsets.zero,
                  child: product.value == null
                      ? MyButton.expandable(
                          height: null,
                          padding: const EdgeInsets.symmetric(vertical: 12),
                          text: 'Crear',
                          fontSize: 20,
                          onPressed: () async {
                            if (productNameController.value.text.isEmpty ||
                                productCodeController.value.text.isEmpty) {
                              ScaffoldMessenger.of(context).showSnackBar(
                                  MySnackBar(
                                      screenSize: MediaQuery.sizeOf(context),
                                      isLandscape: isLandscape,
                                      text: "Faltan campos por rellenar"));

                              return;
                            }
                            if (productPriceController.value.text == '0.00' ||
                                productCodeController.value.text == '0') {
                              ScaffoldMessenger.of(context).showSnackBar(
                                  MySnackBar(
                                      screenSize: MediaQuery.sizeOf(context),
                                      isLandscape: isLandscape,
                                      text:
                                          "Debe introducir una cantidad valida"));

                              return;
                            }
                            productController.createProduct(initProduct());
                          })
                      : Column(
                          children: [
                            MyButton.expandable(
                                height: null,
                                padding:
                                    const EdgeInsets.symmetric(vertical: 12),
                                text: 'Actualizar',
                                fontSize: 20,
                                onPressed: () async {
                                  productController
                                      .updateProduct(initProduct());
                                }),
                            SizedBox(
                              height: 12,
                            ),
                            MyButton.expandable(
                                textColor: AppColors.primaryDark,
                                color: Colors.white,
                                height: null,
                                padding:
                                    const EdgeInsets.symmetric(vertical: 12),
                                text: 'Eliminar',
                                fontSize: 20,
                                onPressed: () async {
                                  showDialog(
                                      context: context,
                                      builder: (context) => DeleteDialog(
                                            onAccept: () async =>
                                                await productController
                                                    .deleteProduct(
                                                        product.value!.code),
                                            text:
                                                '¿Está seguro que desea eliminar éste producto?',
                                          ));
                                })
                          ],
                        ),
                ),
                const SizedBox(height: 10),
              ],
            ),
          ),
        ),
      ),
    );
  });
}
